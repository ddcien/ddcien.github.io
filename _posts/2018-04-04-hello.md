# Hello world

```ruby
def hello(name)
    puts "Hello, #{name}"
end

hello('world')
```

```python
def hello(name):
    print("Hello, {}".format(name))

hello('world')
```

```c
#include <stdio.h>

void hello(char const *name) {
    printf("Hello, %s", name);
}

int main(int argc, char *argv[]) {
    (void)argc;
    (void)argv;
    hello("world");
    return 0;
}
```

```cpp
#include <iostream>

void hello(std::string const &name) {
    std::cout << "Hello " << name << std::endl;
}

int main(int argc, char *argv[]) {
    (void)argc;
    (void)argv;
    hello(std::string("world"));
    return 0;
}
```
